                          ////OPERADORES///

            ARITIMÉTICOS
+  -- SOMA

* -- MULTIPLICAÇÃO
/ -- DIVISÃO
% -- RESTO DA DIVISÃO INTEIRA
** -- POTÊNCIA

------ORDEM DE PRECEDÊNCIA:
() > ** > * / % > + -

------AUTO ATRIBUIÇÃO:

SE QUISER ATRIBUIR UM NOVO VALOR A UMA VARIAVEL JA EXISTENTE, ULTILIZA-SE:
 < var += -novo valor >
Exemplos:

var n = 3

n = n + 4 -- n += 4
n = n - 5 -- n -= 5 
n = n * 4 -- n *= 4 
n = n / 2 --  n /= 2
n = n**2 -- n **= 2
n = n % 5 --  n %= 5

----INCREMENTO:

DA PRA SIMPLIFICAR MAIS AINDA: 

n = 5
n += 1-> n++ 
n -= 1 -> n-- 

OBS: SE EU COLOCAR O -- (OU ++) ANTES DA VARIÁVEL, É CONSIDERADO UM PRÉ INCREMENTO


                       ///OPERADORES RELACIONAIS///

IMPORTANTE DESTACAR QUE TODO OPERADOR RELACIONAL RETORNA UM VALOR BOOLEANO (TRUE OR FALSE)


> : maior que 
< : menor que 
>= : maior ou igual 
<= : menor ou igual
== : igual 
!= : diferente


                       ///OPERADORES LÓGICOS///
 ! : operador de negação, ex:
 true -> false 
 false -> true

&& : so retorna o valor TRUE se os dois valores forem verdadeiros, ex:

true && true -> true
true && false -> false
false && true -> false
false && false -> FALSE

|| retorna o valor TRUE se tiver os dois ou um lado verdadeiro, ex:
true || true -> true 
true || false -> true
false || true  -> true
false || false -> false

---------------------HIERARQUIA DOS OPERADORES RELACIONAIS-----------------------
      ! >> && >> ||
